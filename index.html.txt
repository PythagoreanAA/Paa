<script>
const modal = document.getElementById("modal");
const toast = document.getElementById("toast");
const copyStatus = document.getElementById("copyStatus");

document.getElementById("year").textContent = new Date().getFullYear();

function openModal(){
  modal.style.display = "grid";
  toast.textContent = "";
}
function closeModal(){
  modal.style.display = "none";
  toast.textContent = "";
}
modal.addEventListener("click", (e)=>{
  if(e.target === modal) closeModal();
});
document.addEventListener("keydown", (e)=>{
  if(e.key === "Escape" && modal.style.display === "grid") closeModal();
});

async function copyEmail(){
  const email = document.getElementById("emailText").textContent.trim();
  try{
    await navigator.clipboard.writeText(email);
    toast.textContent = "Copied.";
  }catch{
    toast.textContent = "Copy failed. Select and copy manually.";
  }
}

async function copySiteLink(){
  const link = window.location.href.split("#")[0];
  try{
    await navigator.clipboard.writeText(link);
    copyStatus.textContent = "Link copied.";
    setTimeout(()=> copyStatus.textContent = "", 1400);
  }catch{
    copyStatus.textContent = "Copy failed.";
    setTimeout(()=> copyStatus.textContent = "", 1400);
  }
}
</script>
